validator.js function chkName(){ var myName=document.getElementById("custName"); var pos=myName.value.search(/^[A-Z][a-z]+,?[A-Z][a-z]+,?[A-Z]\.?$/); if(pos!=0) { alert("The name you entered("+myName.value+")is not in the correct form.\n"+"The correct form is:"+"last-name,first-name,middleinitial\n"+"Please go back annd fix your name"); return false; }else return true; } function chkPhone(){ var myPhone=document.getElementById("phone"); var pos=myPhone.value.search(/^\d{3}-\d{3}-\d{4}$/); if(pos!=0) { alert("The phone number you entered("+myPhone.value+")is not in the correct form.\n"+"The correct form is:ddd-ddd-dddd\n"+"Please go back annd fix your phone number"); return false; }else return true; } document.getElementById("custName").onchange=chkName; document.getElementById("phone").onchange=chkPhone;
